# Commands for setting up the `MLMA_Env` Environment on Terminal 

python3 -m venv MLMA_Env
source MLMA_Env/bin/activate
python -m pip install --upgrade pip

# install notebook packages into that env
pip install ipykernel jupyter pandas numpy scikit-learn matplotlib tensorflow

# register environment as a jupyter kernel
python -m ipykernel install --user --name mlma-finalproj --display-name "Python (mlma-finalproj)"
